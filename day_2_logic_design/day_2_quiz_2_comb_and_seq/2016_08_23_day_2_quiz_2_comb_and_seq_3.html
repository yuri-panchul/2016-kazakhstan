The common testbench used to generate the waveforms

        module testbench;
        
            logic clk;
            logic [7:0] d, q;
        
            dut dut (clk, d, q);
        
            initial
            begin
               clk = 0;
        
               forever
                   #10 clk = ! clk;
            end
        
            initial
            begin
                $dumpvars ();
        
                for (int i = 0; i < 10; i++)
                begin
                    @(posedge clk);
                    #15;
                    d = i;
                end
        
                $finish;
            end
        
        endmodule
